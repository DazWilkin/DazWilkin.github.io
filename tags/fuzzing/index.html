Python

atheris-libprotobuf-mutator requires Bazel to build so I used Google Cloud Shell which includes a Bazel installation.

python3 -m venv venv
source venv/bin/activate

python3 -m pip install grpcio
python3 -m pip install grpcio-tools

python3 \
-m grpc_tools.protoc \
--proto_path=${PWD} \
--python_out=${PWD} \
--pyi_out=${PWD} \
--grpc_python_out=${PWD} \
${PWD}/*.proto

python3 -m pip install atheris
python3 -m pip install atheris-libprotobuf-mutator

import atheris
import atheris_libprotobuf_mutator
import sys

import ackal_pb2

@atheris.instrument_func
def TestOneProtoInput(msg):
        print(msg)

if __name__ == "__main__":
        atheris_libprotobuf_mutator.Setup(
                sys.argv,
                TestOneProtoInput,
                proto=ackal_pb2.CheckCreateRequest
        )
        atheris.Fuzz()
